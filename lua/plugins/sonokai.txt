return {
  "sainnhe/sonokai",
  lazy = false,
  priority = 1000,
  config = function()
    -- Set the Sonokai variant (options: default, atlantis, andromeda, shusia, maia, espresso)
    vim.g.sonokai_style = "default"

    -- Optionally enable italics
    vim.g.sonokai_enable_italic = true

    -- Load the colorscheme
    vim.cmd.colorscheme("sonokai")
  end,
}
